vector<long> bitwiseEquations(vector<long>& a, vector<long>& b) {
    vector<long> result;
    
    for (int i = 0; i < a.size(); i++) {
        long ai = a[i];
        long bi = b[i];
        
        bool found_solution = false;
        long x = 0;
        
        while (x <= ai) {
            long y = ai - x;
            if ((x ^ y) == bi) {
                result.push_back(2 * x + 3 * y);
                found_solution = true;
                break;
            }
            x++;
        }
        
        if (!found_solution) {
            result.push_back(0);
        }
    }
    
    return result;
}
